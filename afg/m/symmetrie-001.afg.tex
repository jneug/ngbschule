\begin{aufgabe}[subtitle=Punkte im Koordinatensystem]
	\begin{teilaufgaben}
		\teilaufgabe Lies die \textbf{Koordinaten der Punkte A bis D} ab und notiere sie in der korrekten Schreibweise \textit{in deinem Heft}.
		\teilaufgabe \textbf{Markiere die Punkte E, F und G} im Koordinatensystem \textit{auf dem Arbeitsblatt}. \begin{center}
			\begin{koordinatensystemNO}[5mm]{13}{13}
				\Punkt{2,5}{A}
				\Punkt{6,2}{B}
				\Punkt{7,7}{C}
				\Punkt*{10,0}{D}\PunktBeschriften[above right]{D}{D}
			\end{koordinatensystemNO}\end{center}
	\end{teilaufgaben}

	\begin{loesung}
		\begin{teilaufgaben}
			\teilaufgabe \begin{multicols}{2}
				\begin{smallitemize}
					\item[] $A(2|5)$
					\item[] $B(6|2)$
					\item[] $C(7|7)$
					\item[] $D(10|0)$
				\end{smallitemize}
			\end{multicols}
			\teilaufgabe \begin{center}
				\begin{koordinatensystemNO}[5mm]{13}{13}
					\Punkt{4,9}{E}
					\Punkt{0,11}{F}
					\Punkt{1.5,10.5}{G}
			\end{koordinatensystemNO}\end{center}
		\end{teilaufgaben}
	\end{loesung}

	\begin{erwartungen}
		\erwartung{... notiert die Koordinaten der Punkte $A$, $B$, $C$ und $D$ in korrekter Schreibweise.}{4}
		\erwartung{... markiert die Punkte E und F korrekt im Koordinatensytstem.}{2}
		\erwartung{... markiert den Punkt G korrekt im Koordinatensystem.}{0}[1]
	\end{erwartungen}
\end{aufgabe}
